    sourceType: 'module'

  plugins: [
    'html'

  'rules': {
    'prettier/prettier': ['error'],// [error,wran,off]

    "eol-last": 0,//文件以单一的换行符结束
    "no-extra-semi": 0,//禁止多余的冒号
